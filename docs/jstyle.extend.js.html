  7 //add fieldset style
  8 jStyle.addStyle("fieldset");
  9 jStyle.styles.fieldset.filter="fieldset";
 10 jStyle.styles.fieldset.disabled=false;
 11 jStyle.styles.fieldset.addParameter({name:'collasible'});
 12 jStyle.styles.fieldset.addParameter('closed');//parameters
 13 jStyle.styles.fieldset.render=function(style,element)//render jstyle
 14 		{
 15 			if (!style.collasible) return;
 16 			var lengend_class='clsCollasibleLegend';
 17 			if(style.closed) {
 18 				jStyle.cssSelector(":not(legend):first",element)[0].style.display="none";
 19 				lengend_class='clsCollasibleLegendClosed';
 20 			};
 21 			legend=jStyle.find("legend",element)[0];
 22 			if(!legend) return false;
 23 			cs$(legend).cls("+",lengend_class);
 24 			legend.onclick=function() {
 25                 var lg = $(this);
 26                 var fs = lg.closest("fieldset");
 27 				var contentSpan=fs.children(":not(legend):first");
 28                 var bg = $(this).css("background-image");
 29                 if (bg.indexOf("minus.gif") > 0) {
 30                     contentSpan.slideUp("fast");
 31                     lg.css("background-image", bg.replace("minus", "plus"));
 32                 } else {
 33                     contentSpan.slideDown("fast");
 34                     lg.css("background-image", bg.replace("plus", "minus"));
 35                 }
 36             };//.click();
 37 };//render
